# introduction
- /etc/crontab trigger /opt/etcdbackup.sh once every 3 days
  - run /var/usrlocal/bin/cluster-backup.sh on master
  - create temporary backup on master node:/tmp/.etcdbackup
- bastion scp backupfile from master
- delete temporary file on master
- rotate 7 set of backup under /ETCD.BACKUP/
# pre-request
- ssh to ocp bastion
```bash
ssh root@<BASTION'S IP>
```
# add cronjob
- append job to /etc/crontab
```bash
[bastion ~ ] # cat /etc/crontab
SHELL=/bin/bash
PATH=/sbin:/bin:/usr/sbin:/usr/bin        
MAILTO=root
0 0 */3 * * root bash /opt/etcdbackup.sh
```
# add script
- place etcdbackup.sh under /opt/
```bash
[bastion ~ ] # cat /opt/etcdbackup.sh 
#!/bin/bash
        WM=core@<MASTER_NODE'S_NAME>
        mkdir /ETCD.BACKUP
        cd /ETCD.BACKUP
        ssh $WM << EOF
        mkdir /tmp/.etcdbackup
        sudo /var/usrlocal/bin/cluster-backup.sh /tmp/.etcdbackup
        sudo chown -R core:core /tmp/.etcdbackup
EOF
        scp $VM:/tmp/.etcdbackup/* /ETCD.BACKUP/
        ssh $VM << EOF
        rm -rf /tmp/.etcdbackup
EOF
        ls -t /ETCD.BACKUP |sed -e '1,14d' | xargs rm -rf
```
